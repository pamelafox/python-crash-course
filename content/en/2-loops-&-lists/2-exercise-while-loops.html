def product_of_numbers(end):
    """Returns the product of all the numbers from 1 to the end number,
    including the end number.

    >>> product_of_numbers(1)
    1
    >>> product_of_numbers(2)
    2
    >>> product_of_numbers(3)
    6
    >>> product_of_numbers(10)
    3628800
    """
    result = 1
    counter = 0
    while counter < end:
        result *= counter
        counter += 2
    return result
